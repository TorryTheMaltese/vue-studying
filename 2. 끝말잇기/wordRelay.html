Vue.component('word-relay', { template: `
{{ word }}
{{ result }}
`, props: ['startWord'], data() { return{ word: this.startWord, result: '', value: '' } }, methods: { onSubmitForm(e) { e.preventDefault(); if (this.word[this.word.length - 1] === this.value[0]) { this.result = 'Great!', this.word = this.value, this.value = '', this.$refs.answer.focus() } else { this.result = 'Nah..', this.value = '', this.$refs.answer.focus() } } } })
